const int buttonPin = 4;  //Aqui se indica que el boton esta conectado al pin 4   
const int ledPin =  12; // y que el LED estara conectado al pin 12
int buttonState = 0; // este es el estado del boton, el cual se encuentra en 0, que significa que no esta presionado   
void setup() {
  pinMode(ledPin, OUTPUT); //Aqui se define que el led sera una salida
  pinMode(buttonPin, INPUT); //Y el boton sera una entrada
}
void loop() {
  buttonState = digitalRead(buttonPin); //Se lee el boton y su estado
  if (buttonState == HIGH) { //Si esta el boton presionado, se encendera el Led
    digitalWrite(ledPin, HIGH);
  } 
  else {
    digitalWrite(ledPin, LOW); //Y si no lo esta, se mantendra apagado
  }
  
  delay(50);
}
